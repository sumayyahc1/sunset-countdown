<!DOCTYPE html>

<html>

  <head>

    <title>Countdown to Sunset</title>

    <style>

      #countdown {

        font-size: 24px;

        font-weight: bold;

      }

    </style>

  </head>

  <body>

    <h1>Countdown to Sunset</h1>

    <div id="countdown"></div>


    <script>

      // Obtain sunset time using Sunrise-Sunset.org API

      fetch(

        "https://api.sunrise-sunset.org/json?lat=37.7749&lng=-122.4194&date=today"

      )

        .then((response) => response.json())

        .then((data) => {

          const sunsetTime = new Date(data.results.sunset);

          const now = new Date();

          const timeDiff = sunsetTime - now;


          // Update countdown timer every second

          setInterval(() => {

            const hours = Math.floor(timeDiff / (1000 * 60 * 60));

            const minutes = Math.floor(

              (timeDiff % (1000 * 60 * 60)) / (1000 * 60)

            );

            const seconds = Math.floor((timeDiff % (1000 * 60)) / 1000);


            document.getElementById(

              "countdown"

            ).textContent = `${hours} hours, ${minutes} minutes, ${seconds} seconds until sunset`;


            timeDiff -= 1000;

          }, 1000);

        });

    </script>

  </body>

</html>
